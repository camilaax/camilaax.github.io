MergeSort

Complexidades:

Melhor caso: O(n log n)

Caso médio: O(n log n)

Pior Caso: O(n log n)

Código:

    private static void merge(Integer[] x, int inicio, int fim) {
        int meio;
        if (inicio < fim) {
            meio = (inicio + fim)/2;
            merge(x, inicio, meio);
            merge(x, meio+1, fim);
            intercala(x, inicio, fim, meio);
        }
    }

    private static void intercala(Integer[] x, int inicio, int fim, int meio) {
        int i = inicio, j = meio + 1, k = 0, n = fim-inicio+1;
        Integer[] temp = new Integer[n];

        while (i <= meio || j <= fim) {
            if (i > meio) {
                temp[k++] = x[j++];
            } else if (j > fim) {
                temp[k++] = x[i++];
            } else if (x[i] < x[j]) {
                temp[k++] = x[i++];
            } else {
                temp[k++] = x[j++];
            }
        }

        for (j = 0; j < n; j++) {
            x[inicio + j] = temp[j];
        }
    }

QuickSort

Complexidades:

Melhor caso: O(n²)

Caso médio: O(n log n)

Pior Caso: O(n log n)

Código:

    public static void quicksort(Integer[] x, int inicio, int fim) {
        while(inicio < fim) {
            int particao = particionador(x, inicio, fim);
            if (particao - inicio <= fim - (particao + 1)) {
                quicksort(x, inicio, particao);
                inicio = particao + 1;
            } else {
                quicksort(x, particao + 1, fim);
                fim = particao;
            }
        }
    }

    private static int particionador(Integer[] x, int inicio, int fim) {
        int temp = x[inicio];
        int i = inicio;
        for (int j = inicio + 1; j <= fim; j++)
            if (x[j] < temp) {
            i++;
            trocar(x, i, j);
        }
        trocar(x, inicio, i);
        return i;
    }

    private static void trocar(Integer[] x, int i, int j) {
        int temp = x[i];
        x[i] = x[j];
        x[j] = temp;
    }

HeapSort

Complexidades:

Melhor caso: O(n log n)

Caso médio: O(n log n)

Pior Caso: O(n log n)

Código:

    public static void heapsort(Integer[] x) {
        for (int i = x.length / 2 - 1; i >= 0; i--) {
            deslocar(x, i, x.length);
        }
        for (int i = x.length - 1; i > 0; i--) {
            trocar(x, 0, i);
            deslocar(x, 0, i);
        }
    }

    private static void deslocar(Integer[] x, int i, int n) {
        int aux, temp;

        for (temp = x[i]; esquerda(i) < n; i = aux) {
            aux = esquerda(i);
            if (aux != n - 1 && (x[aux] < x[aux + 1])) {
                aux++;
            }
            if (temp < x[aux]) {
                x[i] = x[aux];
            } else {
                break;
            }
        }
        x[i] = temp;
    }

    private static int esquerda(int i) {
        return 2 * i + 1;
    }

    private static void trocar(Integer[] x, int i, int j) {
        int temp = x[i];
        x[i] = x[j];
        x[j] = temp;
    }

BucketSort

Complexidades:

Melhor caso: O(n²)

Caso médio: O(n + k)

Pior Caso: O(n + k)

Código:

    static Integer[] bucketsort(Integer[] x, int maior) {
        int[] bucket = new int[maior + 1];
        Integer[] vetor_ordenado = new Integer[x.length];

        for (int i = 0; i < x.length; i++) {
            bucket[x[i]]++;
        }

        int posicao = 0;
        for (int i = 0; i < bucket.length; i++) {
            for (int j = 0; j < bucket[i]; j++) {
                vetor_ordenado[posicao++] = i;
            }
        }
        return vetor_ordenado;
    }

    static int maior(Integer[] x) {
        int maior = 0;
        for (int i = 0; i < x.length; i++) {
            if (x[i] > maior) {
                maior = x[i];
            }
        }
        return maior;
    }

    List[] buckets = new ArrayList[10];
    for (int i = 0; i < buckets.length; i++) {
        buckets[i] = new ArrayList();
    }

    boolean flag = false;
    int tmp = -1, divisor = 1;
    while (!flag) {
        flag = true;
        for (Integer i : vetor) {
            tmp = i / divisor;
            buckets[tmp % 10].add(i);
            if (flag && tmp > 0) {
                flag = false;
            }
        }
        int a = 0;
        for (int b = 0; b < 10; b++) {
            for (Integer i : buckets[b]) {
                vetor[a++] = i;
            }
        buckets[b].clear();
        }
        divisor *= 10;
    }

CountingSort

Complexidades:

Melhor caso: O(n + k)

Caso médio: O(n + k)

Pior Caso: O(n + k)

Código:

    public static void countingsort(Integer[] vetor, int tamanho) {
        Integer[] vetor_ordenado = new Integer[tamanho + 1];

        int maior = vetor[0];
        for (int i = 1; i < tamanho; i++) {
            if (vetor[i] > maior) {
                maior = vetor[i];
            }
        }
        int[] contador = new int[maior + 1];

        for (int i = 0; i < maior; ++i) {
            contador[i] = 0;
        }

        for (int i = 0; i < tamanho; i++) {
            contador[vetor[i]]++;
        }

        for (int i = 1; i <= maior; i++) {
            contador[i] += contador[i - 1];
        }

        for (int i = tamanho - 1; i >= 0; i--) {
            vetor_ordenado[contador[vetor[i]] - 1] = vetor[i];
            contador[vetor[i]]--;
        }

        for (int i = 0; i < tamanho; i++) {
            vetor[i] = vetor_ordenado[i];
        }
    }
